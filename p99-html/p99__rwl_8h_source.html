113 P00_FUTEX_INLINE(p99_rwl_rdlock)
114 int p99_rwl_rdlock(p99_rwl volatile* p00_r) {
115  uint p00_res = 0;
116  atomic_fetch_add_explicit(&(p00_r->p00_w), 1u, memory_order_acq_rel);
117  P99_FUTEX_COMPARE_EXCHANGE(&(p00_r->p00_f), p00_act,
118  /* block if there is an exclusive lock or
119  already too many readers */
120  (P00_RWL_DIAG(stderr, "rdlock found %u\n", p00_act),
121  (p00_act < P00_RWL_EXCL-1)),
122  /* as soon as there is no exclusive lock
123  and the number of waiters is not too
124  big add us to the count */
125  (p00_res = ((p00_act < P00_RWL_EXCL-1) ? p00_act + 1U : p00_act)),
126  /* never wakeup anybody */
127  0U, 0U);
128  atomic_fetch_add_explicit(&(p00_r->p00_w), -1u, memory_order_acq_rel);
129  P00_RWL_DIAG(stderr, "rdlock set to %u\n", p00_res);
130  /* Return EAGAIN if we couldn't increment because there were too
131  many waiters. */
132  return (p00_res <= P00_RWL_EXCL-1)
133  ? 0
134  : EAGAIN;
135 }

148 P00_FUTEX_INLINE(p99_rwl_wrlock)
149 int p99_rwl_wrlock(p99_rwl volatile* p00_r) {
150  uint p00_res = 0;
151  atomic_fetch_add_explicit(&(p00_r->p00_w), 1u, memory_order_acq_rel);
152  P99_FUTEX_COMPARE_EXCHANGE(&(p00_r->p00_f), p00_act,
153  /* block if there is any lock */
154  (P00_RWL_DIAG(stderr, "wrlock found %u\n", p00_act), !p00_act),
155  /* as soon as there is no lock, reserve exclusively */
156  (p00_res = P00_RWL_EXCL),
157  /* never wakeup anybody */
158  0U, 0U);
159  atomic_fetch_add_explicit(&(p00_r->p00_w), -1u, memory_order_acq_rel);
160  P00_RWL_DIAG(stderr, "wrlock set to %u\n", p00_res);
161  return 0;
162 }

170 P00_FUTEX_INLINE(p99_rwl_inc_conditionally)
171 int p99_rwl_unlock(p99_rwl volatile* p00_r) {
172  uint p00_res = 0;
173  P99_FUTEX_COMPARE_EXCHANGE(&(p00_r->p00_f), p00_act,
174  /* never block */
175  (P00_RWL_DIAG(stderr, "unlock found %u\n", p00_act), true),
176  /* decrement for shared locks, force to 0 for exclusives */
177  (p00_res = ((p00_act == P00_RWL_EXCL) ? 0U : p00_act - 1U)),
178  /* wakeup potential waiters if any and
179  the count fell to 0 */
180  0U, (p00_res ? 0U : atomic_load(&p00_r->p00_w)));
181  P00_RWL_DIAG(stderr, "unlock set to %u\n", p00_res);
182  return 0;
183 }
