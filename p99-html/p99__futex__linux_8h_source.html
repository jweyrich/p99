242 void p99_futex_wakeup(p99_futex volatile* p00_cntp,
243  unsigned p00_wmin, unsigned p00_wmax) {
244  if (p00_wmax < p00_wmin) p00_wmax = p00_wmin;
245  if (p00_wmax) {
246  unsigned volatile*const p00_cnt = (unsigned*)p00_cntp;
247  static_assert(sizeof *p00_cntp == sizeof *p00_cnt,
248  "linux futex supposes that there is no hidden lock field");
249  for (;;) {
250  register signed p00_wok = p00_futex_wake((int*)p00_cnt, p00_wmax);
251  assert(p00_wok >= 0);
252  if (p00_wok >= p00_wmin) break;
253  p00_wmax -= p00_wok;
254  p00_wmin -= p00_wok;
255  }
256  }
