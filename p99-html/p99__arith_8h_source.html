26 /* The wonders of unsigned types make it act as if it were a signed
27  number in 2complement representation. Unsigned types always have
28  logical shift so we negate the result to simulate an arithmetic
29  shift in that special case.
30 
31  To get back to intmax_t even more wonders occur. The result of
32  P99_UINTMAX_HIGHBIT is 0 or 1 and so it fits into intmax_t. That
33  converted value then can be negated to obtain 0 or -1 as a

93 p99_inline uintmax_t p99_arith_log2(uintmax_t p00_a) {
94  uintmax_t p00_l = 0;
95  /* If P99_UINTMAX_WIDTH is a power of two, things are relatively
96  simple. */
97 #if P99_LOW2(P99_UINTMAX_WIDTH) == P99_UINTMAX_WIDTH
98  P99_CONSTANT(int, p00_ll, P99_HIGH2(P99_UINTMAX_WIDTH) - 1);
99 #else
100  /* If not, we xor out the lowest bit until a power of two
101  remains. This while loop should result in a compile time
102  computation, but who knows. */
103  unsigned p00_ll = P99_UINTMAX_WIDTH;
104  while (P99_LOW2(p00_ll) != p00_ll) p00_ll ^= P99_LOW2(p00_ll);
105  p00_ll = P99_HIGH2(p00_ll) - 1;
106 #endif
107  /* Any decent compiler will unroll this loop */
108  for(int p00_i = p00_ll; p00_i >= 0; --p00_i) {
109  uintmax_t p00_s = UINTMAX_C(1) << p00_i;
110  uintmax_t p00_m = ((UINTMAX_C(1) << p00_s) - 1) << p00_s;
111  uintmax_t p00_t = !!(p00_a & p00_m);
112 
113  p00_l |= p00_t * p00_s;
114  p00_a >>= p00_t * p00_s;
115  }
116  return p00_l;
