478 /* Split the "va_arg" argument of the P99_GETOPT macros into the name
479  of the variable and the default value for that variable. We have to
480  take the raw string produced from that argument such that we see
481  the string as the programmer put it, not as the preprocessor
482  expanded it. */
483 p99_inline
484 void p00_getopt_help_split(size_t p00_len, char p00_buf[p00_len],
485  char const**p00_n, char const** p00_v) {
486  size_t p00_pos = 0;
487  p00_pos += strspn(&p00_buf[p00_pos], " \t\n");
488  *p00_n = &p00_buf[p00_pos];
489  p00_pos += strcspn(p00_buf, ", \t\n");
490  if (p00_buf[p00_pos]) {
491  p00_buf[p00_pos] = 0;
492  ++p00_pos;
493  p00_pos += strspn(&p00_buf[p00_pos], ", \t\n");
494  *p00_v = &p00_buf[p00_pos];
495  p00_pos += strcspn(*p00_v, ", \t\n");
496  p00_buf[p00_pos] = 0;
497  } else *p00_v = "0";
498 }

730  atexit(p00_getopt_atexit);
731  char const* p00_err0 = 0;
732  char const* p00_err1 = 0;
733  char const* p00_err2 = 0;
734  /* Create a sorted array with all the aliases, such that we may then
735  search for a matching key. */
736  static struct p00_getopt const* p00_A[] = {
737  P00_GETOPT_ARRAY_INIT(P00_GETOPT_CHARS),
738  };
739  memset(p00_A, 0, sizeof p00_A);
740  {
741  P00_GETOPT_ARRAY_ASSIGN(P00_GETOPT_CHARS);
742  }
743  qsort_s(p00_A, P99_ALEN(p00_A), sizeof *p00_A, p00_getopt_comp, 0);
744 
745  register struct p00_getopt const** p00_up = p00_A;
746  while (*p00_up) ++p00_up;

748  /* If --help is already used by the application code cancel the
749  mention of --help in p00_getopt_help otherwise insert the alias. */
750  if (p00_getopt_find_alias("help", p00_up-p00_A, p00_A)) {
751  *p00_up = &p00_getopt_help[1];
752  } else {
753  *p00_up = &p00_getopt_help[0];
754  }
755  ++p00_up;
756  qsort_s(p00_A, p00_up - p00_A, sizeof *p00_A, p00_getopt_comp, 0);
757 
758  p00_getopt_progname = (*p00_argv)[0];

760  /* Now comes the main processing loop. One character arguments may
761  be aggregated into one option, that is why this loop looks a bit
762  scary. */
763  int p00_arg = 1;
764  for (; p00_arg < *p00_argc && (*p00_argv)[p00_arg]; ++p00_arg) {
765  /* All options must start with a dash, otherwise this finishes
766  option processing. */
767  if ((*p00_argv)[p00_arg][0] != '-') {
768  goto P00_REARANGE;
769  } else {
770  /* A "--" without anything else finishes option processing. */
771  if ((*p00_argv)[p00_arg][1] == '-' && !(*p00_argv)[p00_arg][2]) {
772  ++p00_arg;
773  goto P00_REARANGE;
774  }
775  for (char const* p00_str = (*p00_argv)[p00_arg] + 1; p00_str && p00_str[0];) {
776  int p00_used = 0;
777  bool p00_extra = false;
778  char p00_C = p00_str[0];
779  ++p00_str;
780  /* If there was nothing left in the option, take the next one
781  as an argument. */
782  if (!p00_str[0]) {
783  p00_str = (*p00_argv)[p00_arg + 1];
784  if (p00_str) p00_extra = true;
785  }
786  static char p00_err[2];
787  p00_err[0] = p00_C;
788  p00_err1 = p00_err;
789  p00_err2 = p00_str;
790  switch (p00_C) {
791  /* The cases for the one-character options are hidden
792  here. */
793  P00_GETOPT_INITIALIZE(P00_GETOPT_CHARS)
794  /* If the initial string was "--" this announces a long
795  option. */
796  case '-': {
797  /* First split up the option string into option and
798  argument, if possible. */
799  char p00_al[strlen(p00_str) + 1];
800  strcpy(p00_al, p00_str);
801  char* p00_ar = strchr(p00_al, '=');
802  if (p00_ar) {
803  *p00_ar = 0;
804  ++p00_ar;
805  } else {
806  /* If not, suppose that the argument has been given in
807  the next option. */
808  p00_ar = (*p00_argv)[p00_arg + 1];
809  p00_extra = true;
810  }
811  p00_err1 = p00_al;
812  p00_err2 = p00_ar;
813  /* Search for a matching alias in the array */
814  register struct p00_getopt const*const p00_p = p00_getopt_find_alias(p00_al, p00_up - p00_A, p00_A);
815  if (p00_p) {
816  void* p00_o = p00_p->p00_o;
817  p00_used = p00_p->p00_f(p00_o, p00_ar);
818  if (p00_used >= 0) break;
819  else p00_getopt_diagnostic("unparsable argument", p00_err1, p00_err2);
820  } else p00_getopt_diagnostic("no or ambiguous option alias", p00_err1, p00_err2);
821  goto P00_REARANGE;
822  }
823  }
824  /* If this used the option argument and the argument was found
825  in the next element of argv, skip that. */
826  if (p00_used > 0) {
827  p00_arg += p00_extra;
828  break;
829  } else
830  /* Otherwise if this was a flag and we tried to obtain from
831  the next element in argv, check that element again, it
832  might be an option, too. */
833  if (p00_extra) break;
834  }
835  }
836  }
837 P00_REARANGE:
838  if (p00_err0) p00_getopt_diagnostic(p00_err0, p00_err1, p00_err2);
839  /* At the end of the processing, shift all unused options down, such
840  that they appear at front in the argv array. */
841  *p00_argc -= (p00_arg - 1);
842  for (int p00_n = 1; p00_n < *p00_argc; ++p00_n) {
843  (*p00_argv)[p00_n] = (*p00_argv)[p00_n + (p00_arg - 1)];
844  }
845 }
