135  struct timeval t;
136  if (gettimeofday(&t, P99_0(struct timezone*))) {
137  errno = 0;
138  return 0;
139  } else {
140  p00_ts->tv_sec = t.tv_sec;

281 p99_inline
282 void p00_timeonce_init(void) {
283  /* Calibrate the monotonic time */
284  mach_timebase_info_data_t p00_tb = P99_INIT;
285  mach_timebase_info(&p00_tb);
286  p00_timebase = p00_tb.numer;
287  p00_timebase /= p00_tb.denom;
288  /* Compute the offset of the monotonic time compared to UTC */
289  /* Nanosec since system start, or something similar. */
290  uint64_t p00_nsec = mach_absolute_time() * p00_timebase;
291  p00_timespec_get(&p00_timeoff_, TIME_UTC);
292  uint64_t const p00_giga = UINT64_C(1000000000);
293  uint64_t p00_epoch = p00_timeoff_.tv_sec * p00_giga + p00_timeoff_.tv_nsec;
294  p00_epoch -= p00_nsec;
295  p00_timeoff_.tv_sec = p00_epoch / p00_giga;
296  p00_timeoff_.tv_nsec = p00_epoch % p00_giga;
297 }

299 p99_inline
300 int timespec_get(struct timespec *p00_ts, int p00_base) {
301  call_once(&p00_timeonce, p00_timeonce_init);
302  uint64_t p00_nsec = mach_absolute_time() * p00_timebase;
303  register uint64_t const p00_giga = UINT64_C(1000000000);
304  p00_ts->tv_sec = p00_nsec / p00_giga;
305  p00_ts->tv_nsec = p00_nsec % p00_giga;
306  if (p00_base != TIME_MONOTONIC) {
307  p00_ts->tv_sec += p00_timeoff->tv_sec;
308  p00_ts->tv_nsec += p00_timeoff->tv_nsec;
309  while (p00_ts->tv_nsec >= p00_giga) {
310  p00_ts->tv_nsec -= p00_giga;
311  ++p00_ts->tv_sec;
312  }
313  }
314  return p00_base;
315 }
