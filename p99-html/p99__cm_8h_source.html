107 p99_inline
108 void p00_cm_signal(p99_cm volatile* p00_cm) {
109  if (p00_cm->waiters) {
110  /* In most of the cases the waiter will already be in the kernel
111  wait when we arrive here: this thread has acquired the lock,
112  done whatever processing and now is releasing the
113  lock. Nevertheless there is a small window, where we may have
114  doubled a waiter thread. Therefore we force a busy wait if
115  necessary, until we have woken up someone. */
116  p99_futex_wakeup(&p00_cm->p00_c, 1u, 1u);
117  --p00_cm->waiters;
118  }
119 }
