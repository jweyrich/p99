225 void p00_foreign_cleanup(void) {
226  size_t p00_foreign = atomic_load_explicit(&p00_foreign_nb, memory_order_consume);
227  p00_thrd ** p00_thrd = p00_foreign_tab;
228  p00_foreign_tab = 0;
229  if (p00_foreign) {
230  /* Ensure that all data is synchronized with all threads, not only
231  with the last one that changed p00_foreign_nb. */
232  atomic_thread_fence(memory_order_seq_cst);
233  for (size_t p00_i = 0; p00_i < p00_foreign; ++p00_i) {

235  fputs("found foreign thread\n", stderr);
236  free(p00_thrd[p00_i]);

249  p00_thrd * p00_loc = P00_THRD_LOCAL;
250  if (P99_UNLIKELY(!p00_loc)) {
251  size_t p00_nb = atomic_fetch_add_explicit(&p00_foreign_nb, 1, memory_order_acq_rel);
252  if (!p00_nb) atexit(p00_foreign_cleanup);
253  if ((p00_nb^(p00_nb-1)) == (p00_nb+(p00_nb-1))) {
254  p00_foreign_tab = realloc(p00_foreign_tab, sizeof(p00_thrd*[2*(p00_nb+1)]));
255  }

257  p00_foreign_tab[p00_nb] = p00_loc;
258  *p00_loc = (p00_thrd) {
259  .p00_id = pthread_self(),
260  .p00_foreign = p00_nb + 1,

410  if (P99_UNLIKELY(p00_ret)) return thrd_error;
411  p00_ret = pthread_mutex_init(&P99_ENCP(p00_mtx), &p00_attr);
412  if (P99_UNLIKELY(p00_ret)) return thrd_error;

415  if (P99_UNLIKELY(p00_ret)) return thrd_error;

527 p99_inline
528 int thrd_detach(thrd_t p00_thr) {
529  /* The thread is not yet detached so its pthread id is still
530  valid. If it already has finished, this will just free the
531  resources that pthread holds for it. */
532  int p00_ret = pthread_detach(P99_ENC(p00_thr)->p00_id) ? thrd_error : thrd_success;
533  if (atomic_flag_test_and_set_explicit(&P99_ENC(p00_thr)->p00_detached, memory_order_acq_rel)) {
534  /* The thread has already finished. Free the state, since nobody
535  will join it, anyhow. */
536  free(P99_ENC(p00_thr));
537  return thrd_success;
538  } else {
539  return p00_ret;
540  }
